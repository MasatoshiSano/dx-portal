setup.sh (初回セットアップ)

cat > ~/.dx-config.yml << 'EOF'
brain: ~/dx-brain
portal: ~/Apps/dx-portal
company_data: ~/Apps/dx-portal/company-data
EOF

# dx-brain初期化 (初回のみ)
cp -r ~/Apps/dx-portal/templates/brain ~/dx-brain
